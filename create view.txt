CREATE VIEW v_karyawan AS SELECT karyawan.id, karyawan.nik, karyawan.nama, karyawan.jabatan_id, jabatan.jabatan, karyawan.username, karyawan.password, karyawan.role_id, role.role, karyawan.foto FROM karyawan JOIN jabatan ON karyawan.jabatan_id = jabatan.id JOIN role ON karyawan.role_id = role.id

CREATE VIEW v_approval_list AS SELECT approval_list.id, approval_list.nama_project, approval_list.initiate, karyawan.nama, approval_list.pimpro, karyawan.nama AS nama_pimpro, approval_list.created_at, approval_list.delete_status, approval_list.finish FROM approval_list JOIN karyawan ON approval_list.initiate = karyawan.nik

CREATE VIEW v_project AS SELECT project.id, project.kd_project, project.kd_forecast, project.nik_sales, karyawan.nama, project.kd_badan, badan.nama AS nama_badan, project.nm_project, project.no_ticket, project.no_sph, project.no_spk, project.nm_customer, project.nohp, project.perusahaan, project.kapasitas, project.satuan_kapasitas, project.jumlah_order, project.satuan, project.lokasi_project, project.tipe_project, kategoripenjualan.nm_kategori, project.start, project.deadline, project.nilai_project, project.ppn, project.cashback, project.remark, project.created_at FROM project JOIN karyawan ON project.nik_sales = karyawan.nik JOIN badan ON project.kd_badan = badan.kd_badan JOIN kategoripenjualan ON project.tipe_project = kategoripenjualan.kd_kategori

CREATE VIEW v_pelanggaran AS SELECT pelanggaran.id, pelanggaran.nik, pelanggaran.tanggal, pelanggaran.pelanggaran_id, pelanggaran_list.nama_pelanggaran, pelanggaran_list.status_pelanggaran, pelanggaran.keterangan, pelanggaran.creation_date FROM pelanggaran JOIN pelanggaran_list ON pelanggaran.pelanggaran_id = pelanggaran_list.id
